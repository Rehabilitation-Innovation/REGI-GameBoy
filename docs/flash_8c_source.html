72static int read(blockdevice_t *device, const void *buffer, bd_size_t addr, bd_size_t size) {
73 blockdevice_flash_config_t *config = device->config;
74
75 mutex_enter_blocking(&config->_mutex);
76 const uint8_t *flash_contents = (const uint8_t *)(XIP_BASE + flash_target_offset(device) + (size_t)addr);
77 memcpy((uint8_t *)buffer, flash_contents, (size_t)size);
78 mutex_exit(&config->_mutex);
79
80 return BD_ERROR_OK;

125blockdevice_t *blockdevice_flash_create(uint32_t start, size_t length) {
126 assert(start % FLASH_SECTOR_SIZE == 0);
127 assert(length % FLASH_SECTOR_SIZE == 0);
128
129 blockdevice_t *device = calloc(1, sizeof(blockdevice_t));
130 if (device == NULL) {
131 fprintf(stderr, "blockdevice_flash_create: Out of memory\n");
132 return NULL;
133 }
134 blockdevice_flash_config_t *config = calloc(1, sizeof(blockdevice_flash_config_t));
135 if (config == NULL) {
136 fprintf(stderr, "blockdevice_flash_create: Out of memory\n");
137 free(device);
138 return NULL;
139 }
140
141 device->init = init;
142 device->deinit = deinit;
143 device->read = read;
144 device->erase = erase;
145 device->program = program;
146 device->trim = trim;
147 device->sync = sync;
148 device->size = size;
149 device->read_size = 1;
150 device->erase_size = FLASH_SECTOR_SIZE; // 4096 byte
151 device->program_size = FLASH_PAGE_SIZE; // 256 byte
152 device->name = DEVICE_NAME;
153 device->is_initialized = false;
154
155 config->start = start;
156 config->length = length > 0 ? length : (PICO_FLASH_SIZE_BYTES - start);
157 mutex_init(&config->_mutex);
158 device->config = config;
159 device->init(device);
160 return device;
161}
