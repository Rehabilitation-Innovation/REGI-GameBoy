31static float read_onboard_temperature(const char unit) {
32 const float conversionFactor = 3.3f / (1 << 12);
33
34 float adc = (float)adc_read() * conversionFactor;
35 float tempC = 27.0f - (adc - 0.706f) / 0.001721f;
36
37 if (unit == 'C') {
38 return tempC;
39 } else if (unit == 'F') {
40 return tempC * 9 / 5 + 32;
41 }
42
43 return -1.0f;
44}

46static bool __no_inline_not_in_flash_func(bootsel_button_get)(void) {
47 const uint CS_PIN_INDEX = 1;
48 uint32_t flags = save_and_disable_interrupts();
49 hw_write_masked(&ioqspi_hw->io[CS_PIN_INDEX].ctrl,
50 GPIO_OVERRIDE_LOW << IO_QSPI_GPIO_QSPI_SS_CTRL_OEOVER_LSB,
51 IO_QSPI_GPIO_QSPI_SS_CTRL_OEOVER_BITS);
52 for (volatile int i = 0; i < 1000; ++i);
53 bool button_state = !(sio_hw->gpio_hi_in & (1u << CS_PIN_INDEX));
54 hw_write_masked(&ioqspi_hw->io[CS_PIN_INDEX].ctrl,
55 GPIO_OVERRIDE_NORMAL << IO_QSPI_GPIO_QSPI_SS_CTRL_OEOVER_LSB,
56 IO_QSPI_GPIO_QSPI_SS_CTRL_OEOVER_BITS);
57 restore_interrupts(flags);
58
59 return button_state;
60}

128static void logging_task(void) {
129 if (!enable_logging_task)
130 return;
131
132 static uint64_t last_measure = 0;
133 uint64_t now = time_us_64();
134 uint64_t duration = now - last_measure;
135 if (duration < 1000000)
136 return;
137
138 float temperature = read_onboard_temperature(TEMPERATURE_UNITS);
139 printf("temperature=%.1f\n", temperature);
140
141 int fd = open("/internal/TEMP.TXT", O_WRONLY|O_APPEND|O_CREAT);
142 if (fd == -1) {
143 printf("open('/internal/TEMP.TXT') error: %s\n", strerror(errno));
144 }
145 char buffer[512] = {0};
146 int length = sprintf(buffer, "temperature,%.1f\n", temperature);
147 ssize_t write_size = write(fd, buffer, length);
148 if (write_size == -1) {
149 printf("write error: %s\n", strerror(errno));
150 }
151 int err = close(fd);
152 if (err == -1) {
153 printf("close error: %s\n", strerror(errno));
154 }
155 last_measure = now;
156}

162static void filesystem_management_task(void) {
163 static bool last_status = false;
164
165 bool usb = is_usb_connected();
166 // Override USB connection status as long as the BOOTSEL button is pressed.
167 if (bootsel_button_get())
168 usb = false;
169
170 if (last_status != usb) {
171 if (usb) {
172 enable_logging_task = false;
173 printf("USB connected\n");
174 export_filesystem();
175 filesystem_is_exported = true;
176 } else {
177 printf("USB disconnected\n");
178 filesystem_is_exported = false;
179 enable_logging_task = true;
180 }
181 last_status = usb;
182 }
183}
