24static bool network_init(ip_addr_t *mqtt_server) {
25 if (cyw43_arch_init()) {
26 printf("Wi-Fi init failed");
27 return false;
28 }
29 cyw43_arch_enable_sta_mode();
30
31 while (cyw43_arch_wifi_connect_timeout_ms(WIFI_SSID, WIFI_PASSWORD, CYW43_AUTH_WPA2_AES_PSK, 10000) != 0) {
32 printf("Wi-Fi connection failed\n");
33 sleep_ms(1000);
34 }
35 printf("Wi-Fi connect ok\n");
36
37 while (!ntp_sync()) {
38 printf("NTP sync failed\n");
39 sleep_ms(1000);
40 }
41 printf("NTP sync ok\n");
42
43 while (1) {
44 cyw43_arch_lwip_begin();
45 int err = dns_gethostbyname(MQTT_SERVER, mqtt_server, NULL, NULL);
46 cyw43_arch_lwip_end();
47 if (err == ERR_OK) {
48 while (mqtt_server->addr == 0)
49 sleep_ms(1);
50 break;
51 }
52 printf("lookup %s failed=%d\n", MQTT_SERVER, err);
53 sleep_ms(1000);
54 }
55 printf("lookup %s ok\n", MQTT_SERVER);
56 return true;
57}

69static bool mantain_network_connection(mqtt_client_t *client, ip_addr_t *server) {
70 bool wifi_status = netif_is_up(&cyw43_state.netif[0]) && netif_is_link_up(&cyw43_state.netif[0]);
71 if (!wifi_status) {
72
73 cyw43_arch_lwip_begin();
74 cyw43_wifi_leave(&cyw43_state, CYW43_ITF_STA);
75 int err = cyw43_arch_wifi_connect_timeout_ms(WIFI_SSID, WIFI_PASSWORD, CYW43_AUTH_WPA2_AES_PSK, 1000);
76 cyw43_arch_lwip_end();
77
78 if (err != 0) {
79 printf("Wi-Fi connection failed: err=%d\n", err);
80 return false;
81 }
82 printf("Wi-Fi connect ok\n");
83 }
84 if (!mqtt_client_is_connected(client)) {
85 static struct mqtt_connect_client_info_t ci = {0};
86 ci.client_id = "";
87 ci.client_user = MQTT_USER;
88 ci.client_pass = MQTT_PASSWORD;
89 ci.keep_alive = 30;
90 if (ci.tls_config == NULL)
91 ci.tls_config = altcp_tls_create_config_client(NULL, 0);
92 if (ci.tls_config == NULL) {
93 printf("mantain_network_connection: altcp_tls_create_config_client failed\n");
94 return false;
95 }
96
97 cyw43_arch_lwip_begin();
98 mqtt_client_connect(client, server, MQTT_TLS_PORT, mqtt_connection_cb, 0, &ci);
99 cyw43_arch_lwip_end();
100 return true;
101 }
102 return true;
103}

106static float read_sensor_data() {
107 uint16_t raw = adc_read();
108 const float conversion_factor = 3.3f / (1 << 12);
109 float voltage = raw * conversion_factor;
110 float temperature = 27.0f - (voltage - 0.706f) / 0.001721f;
111 return temperature;
112}

114static void publish_message(mqtt_client_t *client, float data, const char *timestamp) {
115 (void)client;
116 char payload[100] = {0};

118
119 cyw43_arch_lwip_begin();
120 mqtt_publish(client, MQTT_TOPIC, payload, strlen(payload), MQTT_QOS_AT_LEAST_ONCE, 0, NULL, NULL);
121 cyw43_arch_lwip_end();
122
123 printf("Publish: %s\n", payload);
124}

126static void publish_message_from_file(mqtt_client_t *client) {
127 FILE *fp = fopen(LOCAL_QUEUE_PATH, "r");
128 if (fp == NULL) {
129 return;
130 }
131
132 char timestamp[30] = {0};
133 float data = 0;
134 while (fscanf(fp, "%[^,],%f\n", timestamp, &data) != EOF) {
135 publish_message(client, data, timestamp);
136 }
137 fclose(fp);
138 remove(LOCAL_QUEUE_PATH);
139}

152int main(void) {
153 stdio_init_all();
154 rtc_init();
155 adc_init();
156 adc_set_temp_sensor_enabled(true);
157 adc_select_input(4);
158
159 ip_addr_t mqtt_server_addr = {0};
160 if (!network_init(&mqtt_server_addr)) {
161 printf("network_init failed\n");
162 }
163 mqtt_client_t *client = mqtt_client_new();
164 while (1) {
165 float sensor_value = read_sensor_data();
166 const char *timestamp = get_timestamp();
167 if (mantain_network_connection(client, &mqtt_server_addr)) {
168 // Normal operation
169 publish_message_from_file(client);
170 publish_message(client, sensor_value, timestamp);
171 } else {
172 // Network outage operation
173 save_data_to_file(sensor_value, timestamp);
174 }
175 sleep_ms(10 * 1000);
176 }
177
178 mqtt_client_free(client);
179 cyw43_arch_deinit();
180 return 0;
181}
