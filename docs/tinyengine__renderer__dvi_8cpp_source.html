594 count_of(vactive_line_header) * sizeof(uint32_t));
595 }
596
597 // Ensure HSTX FIFO is clear
598 reset_block_num(RESET_HSTX);
599 sleep_us(10);
600 unreset_block_num_wait_blocking(RESET_HSTX);
601 sleep_us(10);
602
603 // Configure HSTX's TMDS encoder for RGB888
604 hstx_ctrl_hw->expand_tmds =
605 7 << HSTX_CTRL_EXPAND_TMDS_L2_NBITS_LSB |
606 16 << HSTX_CTRL_EXPAND_TMDS_L2_ROT_LSB |
607 7 << HSTX_CTRL_EXPAND_TMDS_L1_NBITS_LSB |
608 8 << HSTX_CTRL_EXPAND_TMDS_L1_ROT_LSB |
609 7 << HSTX_CTRL_EXPAND_TMDS_L0_NBITS_LSB |
610 0 << HSTX_CTRL_EXPAND_TMDS_L0_ROT_LSB;
611
612 // Pixels and control symbols (RAW) are an
613 // entire 32-bit word.
614 hstx_ctrl_hw->expand_shift =
615 1 << HSTX_CTRL_EXPAND_SHIFT_ENC_N_SHIFTS_LSB |
616 0 << HSTX_CTRL_EXPAND_SHIFT_ENC_SHIFT_LSB |
617 1 << HSTX_CTRL_EXPAND_SHIFT_RAW_N_SHIFTS_LSB |
618 0 << HSTX_CTRL_EXPAND_SHIFT_RAW_SHIFT_LSB;
619
620 // Serial output config: clock period of 5 cycles, pop from command
621 // expander every 5 cycles, shift the output shiftreg by 2 every cycle.
622 hstx_ctrl_hw->csr = 0;
623 hstx_ctrl_hw->csr =
624 HSTX_CTRL_CSR_EXPAND_EN_BITS |
625 5u << HSTX_CTRL_CSR_CLKDIV_LSB |
626 5u << HSTX_CTRL_CSR_N_SHIFTS_LSB |
627 2u << HSTX_CTRL_CSR_SHIFT_LSB |
628 HSTX_CTRL_CSR_EN_BITS;
629
630
631 // HSTX outputs 0 through 7 appear on GPIO 12 through 19.
632 int HSTX_FIRST_PIN = 12;
633 int clk_p = 12;
634 int rgb_p[3] = { 14, 16, 18 };
635
636 // Assign clock pair to two neighbouring pins:
637 {
638 int bit = clk_p - HSTX_FIRST_PIN;
639 hstx_ctrl_hw->bit[bit] = HSTX_CTRL_BIT0_CLK_BITS;
640 hstx_ctrl_hw->bit[bit ^ 1] = HSTX_CTRL_BIT0_CLK_BITS | HSTX_CTRL_BIT0_INV_BITS;
641 }
642
643 for (uint lane = 0; lane < 3; ++lane)
644 {
645 // For each TMDS lane, assign it to the correct GPIO pair based on the
646 // desired pinout:
647 int bit = rgb_p[lane] - HSTX_FIRST_PIN;
648 // Output even bits during first half of each HSTX cycle, and odd bits
649 // during second half. The shifter advances by two bits each cycle.
650 uint32_t lane_data_sel_bits =
651 (lane * 10) << HSTX_CTRL_BIT0_SEL_P_LSB |
652 (lane * 10 + 1) << HSTX_CTRL_BIT0_SEL_N_LSB;
653 // The two halves of each pair get identical data, but one pin is inverted.
654 hstx_ctrl_hw->bit[bit] = lane_data_sel_bits;
655 hstx_ctrl_hw->bit[bit ^ 1] = lane_data_sel_bits | HSTX_CTRL_BIT0_INV_BITS;
656 }
657
658 for (int i = 12; i <= 19; ++i)
659 {
660 gpio_set_function(i, GPIO_FUNC_HSTX);
661 gpio_set_drive_strength(i, GPIO_DRIVE_STRENGTH_4MA);
662 // if (252000 > 900000) {
663 gpio_set_slew_rate(i, GPIO_SLEW_RATE_FAST);
664 // }
665 }
666
667 printf("GPIO Done\r\n");
668
669 // The channels are set up identically, to transfer a whole scanline and
670 // then chain to the next channel. Each time a channel finishes, we
671 // reconfigure the one that just finished, meanwhile the other channel(s)
672 // are already making progress.
673 // Using just 2 channels was insufficient to avoid issues with the IRQ.
674 dma_channel_config c;
675 c = dma_channel_get_default_config(0);
676 channel_config_set_chain_to(&c, 1);
677 channel_config_set_dreq(&c, DREQ_HSTX);
678 dma_channel_configure(
679 0,
680 &c,
681 &hstx_fifo_hw->fifo,
682 vblank_line_vsync_off,
683 count_of(vblank_line_vsync_off),
684 false
685 );
686 c = dma_channel_get_default_config(1);
687 channel_config_set_chain_to(&c, 2);
688 channel_config_set_dreq(&c, DREQ_HSTX);
689 dma_channel_configure(
690 1,
691 &c,
692 &hstx_fifo_hw->fifo,
693 vblank_line_vsync_off,
694 count_of(vblank_line_vsync_off),
695 false
696 );
697 for (int i = 2; i < NUM_CHANS; ++i)
698 {
699 c = dma_channel_get_default_config(i);
700 channel_config_set_chain_to(&c, (i + 1) % NUM_CHANS);
701 channel_config_set_dreq(&c, DREQ_HSTX);
702 dma_channel_configure(
703 i,
704 &c,
705 &hstx_fifo_hw->fifo,
706 vblank_line_vsync_off,
707 count_of(vblank_line_vsync_off),
708 false
709 );
710 }
711
712 printf("DMA channels claimed\r\n");
713
714 dma_hw->intr = (1 << NUM_CHANS) - 1;
715 dma_hw->ints2 = (1 << NUM_CHANS) - 1;
716 dma_hw->inte2 = (1 << NUM_CHANS) - 1;
717
718 irq_set_exclusive_handler(DMA_IRQ_2, dma_irq_handler);
719 irq_set_enabled(DMA_IRQ_2, true);
720
721 dma_channel_start(0);
722
723 printf("DVHSTX started\r\n");
724
725 return TINYENGINE_OK;
726}
