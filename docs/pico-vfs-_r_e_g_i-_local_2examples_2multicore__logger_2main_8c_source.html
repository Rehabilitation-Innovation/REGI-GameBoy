42static float normal_random(float mean, float stddev) {
43 static bool has_spare = false;
44 static float spare;
45
46 if (has_spare) {
47 has_spare = false;
48 return mean + stddev * spare;
49 }
50
51 has_spare = true;
52 float u, v, s;
53 do {
54 u = (float)rand() / RAND_MAX * 2.0f - 1.0f;
55 v = (float)rand() / RAND_MAX * 2.0f - 1.0f;
56 s = u * u + v * v;
57 } while (s >= 1.0f || s == 0.0f);
58
59 s = sqrtf(-2.0f * logf(s) / s);
60 spare = v * s;
61 return mean + stddev * u * s;
62}
