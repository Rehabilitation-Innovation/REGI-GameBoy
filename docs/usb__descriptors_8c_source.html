3/* A combination of interfaces must have a unique product id, since PC will save device driver after the first plug.
4 * Same VID/PID with different interface e.g MSC (first), then CDC (later) will possibly cause system error on PC.
5 *
6 * Auto ProductID layout's Bitmap:
7 * [MSB] HID | MSC | CDC [LSB]
8 */
9#define _PID_MAP(itf, n) ( (CFG_TUD_##itf) << (n) )
10#define USB_PID (0x4000 | _PID_MAP(CDC, 0) | _PID_MAP(MSC, 1) | _PID_MAP(HID, 2) | \
11 _PID_MAP(MIDI, 3) | _PID_MAP(VENDOR, 4) )

225static uint16_t _desc_str[32];
226
227// Invoked when received GET STRING DESCRIPTOR request
228// Application return pointer to descriptor, whose contents must exist long enough for transfer to complete
229uint16_t const* tud_descriptor_string_cb(uint8_t index, uint16_t langid)
230{
231 (void) langid;
232
233 uint8_t chr_count;
234
235 if ( index == 0)
236 {
237 memcpy(&_desc_str[1], string_desc_arr[0], 2);
238 chr_count = 1;
239 }else
240 {
241 // Note: the 0xEE index string is a Microsoft OS 1.0 Descriptors.
242 // https://docs.microsoft.com/en-us/windows-hardware/drivers/usbcon/microsoft-defined-usb-descriptors
243
244 if ( !(index < sizeof(string_desc_arr)/sizeof(string_desc_arr[0])) ) return NULL;
245
246 const char* str = string_desc_arr[index];
247
248 // Cap at max char
249 chr_count = (uint8_t) strlen(str);
250 if ( chr_count > 31 ) chr_count = 31;
251
252 // Convert ASCII string into UTF-16
253 for(uint8_t i=0; i<chr_count; i++)
254 {
255 _desc_str[1+i] = str[i];
256 }
257 }
258
259 // first byte is length (including header), second byte is string type
260 _desc_str[0] = (uint16_t) ((TUSB_DESC_STRING << 8 ) | (2*chr_count + 2));
261
262 return _desc_str;
263}
