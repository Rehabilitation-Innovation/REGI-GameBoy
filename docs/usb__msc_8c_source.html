103int32_t tud_msc_write10_cb(uint8_t lun, uint32_t lba, uint32_t offset, uint8_t *buffer, uint32_t bufsize) {
104 (void)lun;
105 (void)offset;
106
107 printf(ANSI_MAGENTA "USB write#%lu bufsize=%lu\n" ANSI_CLEAR, lba, bufsize);
108
109 uint32_t block_size = flash2->erase_size;
110 int err = flash2->erase(flash2, lba * block_size, bufsize);
111 if (err != BD_ERROR_OK) {
112 printf("erase error=%d\n", err);
113 }
114 err = flash2->program(flash2, buffer, lba * block_size, bufsize);
115 if (err != BD_ERROR_OK) {
116 printf("program error=%d\n", err);
117 }
118
119 return (int32_t)bufsize;
