58static inline uint32_t urgb_u32(uint8_t r, uint8_t g, uint8_t b)
59{
60 return
61 ((uint32_t)(r) << 8) |
62 ((uint32_t)(g) << 16) |
63 (uint32_t)(b);
64}

66static inline uint32_t urgbw_u32(uint8_t r, uint8_t g, uint8_t b, uint8_t w)
67{
68 return
69 ((uint32_t)(r) << 8) |
70 ((uint32_t)(g) << 16) |
71 ((uint32_t)(w) << 24) |

75void pattern_snakes(PIO pio, uint sm, uint len, uint t)
76{
77 for (uint i = 0; i < len; ++i)
78 {
79 uint x = (i + (t >> 1)) % 64;
80 if (x < 10)
81 put_pixel(pio, sm, urgb_u32(0xff, 0, 0));
82 else if (x >= 15 && x < 25)
83 put_pixel(pio, sm, urgb_u32(0, 0xff, 0));
84 else if (x >= 30 && x < 40)
85 put_pixel(pio, sm, urgb_u32(0, 0, 0xff));
86 else
87 put_pixel(pio, sm, 0);
88 }
89}

91void pattern_random(PIO pio, uint sm, uint len, uint t)
92{
93 if (t % 8)
94 return;
95 for (uint i = 0; i < len; ++i)
96 put_pixel(pio, sm, urgbw_u32(rand() / 100000, rand() / 100000, rand() / 100000, rand() / 100000));

107void pattern_greys(PIO pio, uint sm, uint len, uint t)
108{
109 uint max = 100; // let's not draw too much current!
110 t %= max;
111 for (uint i = 0; i < len; ++i)
112 {
113 put_pixel(pio, sm, t * 0x10101);
114 if (++t >= max) t = 0;
115 }
