47static int _error_remap(int err) {
48 if (err >= 0) {
49 errno = 0;
50 return err;
51 }
52
53 errno = -err;

57static const char *remove_prefix(const char *str, const char *prefix) {
58 size_t len_prefix = strlen(prefix);
59 size_t len_str = strlen(str);
60
61 if (len_str < len_prefix || strncmp(str, prefix, len_prefix) != 0) {
62 return str;
63 }
64 return str + len_prefix;
65}

67static mountpoint_t *find_mountpoint(const char *path) {
68 mountpoint_t *longest_match = NULL;
69 size_t longest_length = 0;
70
71 for (size_t i = 0; i < FS_MAX_MOUNTPOINT; i++) {
72 size_t prefix_length = strlen(mountpoints[i].dir);
73 if (prefix_length > longest_length && strncmp(path, mountpoints[i].dir, prefix_length) == 0) {
74 longest_match = &mountpoints[i];
75 longest_length = prefix_length;
76 }
77 }
78 return longest_match;
79}

98int fs_mount(const char *dir, filesystem_t *fs, blockdevice_t *device) {
99 if (!device->is_initialized) {
100 int err = device->init(device);
101 if (err)
102 return _error_remap(err);
103 }
104
105 int err = fs->mount(fs, device, false);
106 if (err) {
107 return _error_remap(err);
108 }
109
110 auto_init_recursive_mutex(_mutex);
111 recursive_mutex_enter_blocking(&_mutex);
112 for (size_t i = 0; i < FS_MAX_MOUNTPOINT; i++) {
113 if (mountpoints[i].filesystem == NULL) {
114 mountpoints[i].filesystem = fs;
115 mountpoints[i].device = device;
116 mountpoints[i].dir = strdup(dir);
117 recursive_mutex_exit(&_mutex);
118 return _error_remap(0);
119 }
120 }
121 recursive_mutex_exit(&_mutex);
122 return _error_remap(-EFAULT);
123}

275int _fstat(int fildes, struct stat *st) {
276 auto_init_recursive_mutex(_mutex);
277 recursive_mutex_enter_blocking(&_mutex);
278
279 if (fildes == STDIN_FILENO || fildes == STDOUT_FILENO || fildes == STDERR_FILENO) {
280 recursive_mutex_exit(&_mutex);
281 st->st_size = 0;
282 st->st_mode = S_IFCHR;
283 return _error_remap(0);
284 }
285 if (!is_valid_file_descriptor(fildes)) {
286 recursive_mutex_exit(&_mutex);
287 return _error_remap(-EBADF);
288 }
289
290 fs_file_t *file = file_descriptor[FILENO_INDEX(fildes)].file;
291 filesystem_t *fs = file_descriptor[FILENO_INDEX(fildes)].filesystem;
292 if (fs == NULL) {
293 recursive_mutex_exit(&_mutex);
294 return _error_remap(-EBADF);
295 }
296
297 off_t size = 0;
298 if (fs->type != FILESYSTEM_TYPE_FAT) {
299 off_t current = fs->file_tell(fs, file);
300 size = fs->file_seek(fs, file, 0, SEEK_END);
301 off_t err = fs->file_seek(fs, file, current, SEEK_SET);
302 if (current != err) {
303 recursive_mutex_exit(&_mutex);
304 return _error_remap(err);
305 }
306 } else {
307 /* NOTE: Support for different behaviour of FatFs from POSIX
308 *
309 * FatFs has a problem where f_size() reports a larger than actual file size when
310 * moved to a position beyond the f_lseek() file size; POSIX reports the actual size
311 * written to the file, not the seek position.
312 */
313 const char *path = file_descriptor[FILENO_INDEX(fildes)].path;
314 mountpoint_t *mp = find_mountpoint(path);
315 if (mp == NULL) {
316 recursive_mutex_exit(&_mutex);
317 return _error_remap(-ENOENT);
318 }
319 const char *entity_path = remove_prefix(path, mp->dir);
320 filesystem_t *fs = mp->filesystem;
321
322 struct stat finfo = {0};
323 int err = fs->stat(fs, entity_path, &finfo);
324 if (err != 0) {
325 recursive_mutex_exit(&_mutex);
326 return _error_remap(err);
327 }
328 size = finfo.st_size;
329 }
330 recursive_mutex_exit(&_mutex);
331
332 st->st_size = size;
333 st->st_mode = S_IFREG;
334 return _error_remap(0);
335}

337static int _assign_file_descriptor() {
338 int fd = -1;
339 if (max_file_descriptor == 0) {
340 max_file_descriptor = 2;
341 file_descriptor = calloc(max_file_descriptor, sizeof(file_descriptor_t));
342 if (file_descriptor == NULL) {
343 printf("_open: Out of memory\n");
344 return -1;
345 }
346 }
347 for (size_t i = 0; i < max_file_descriptor; i++) {
348 if (file_descriptor[i].filesystem == NULL) {
349 fd = i;
350 break;
351 }
352 }
353 if (fd == -1) {
354 // Extend the management array
355 size_t last_max = max_file_descriptor;
356 max_file_descriptor *= 2;
357 file_descriptor = realloc(file_descriptor, sizeof(file_descriptor_t) * max_file_descriptor);
358 if (file_descriptor == NULL) {
359 printf("_open: Out of memory\n");
360 return -1;
361 }
362 for (size_t i = last_max; i < max_file_descriptor; i++) {
363 file_descriptor[i].filesystem = NULL;
364 file_descriptor[i].file = NULL;
365 }
366 fd = last_max;
367 }
368 return FILENO_VALUE(fd);
369}

405int _open(const char *path, int oflags, ...) {
406 auto_init_recursive_mutex(_mutex);
407 recursive_mutex_enter_blocking(&_mutex);
408
409 mountpoint_t *mp = find_mountpoint(path);
410 if (mp == NULL) {
411 recursive_mutex_exit(&_mutex);
412 return _error_remap(-ENOENT);
413 }
414 const char *entity_path = remove_prefix(path, mp->dir);
415 // find file descriptor
416 int fd = _assign_file_descriptor();
417 if (fd == -1) {
418 recursive_mutex_exit(&_mutex);
419 return _error_remap(-ENFILE);
420 }
421
422 filesystem_t *fs = mp->filesystem;
423 fs_file_t *file = file_descriptor[FILENO_INDEX(fd)].file = calloc(1, sizeof(fs_file_t));
424 if (file == NULL) {
425 recursive_mutex_exit(&_mutex);
426 return _error_remap(-ENOMEM);
427 }
428
429 int err = fs->file_open(fs, file, entity_path, oflags);
430 if (err < 0) {
431 free(file);
432 file_descriptor[FILENO_INDEX(fd)].file = NULL;
433 recursive_mutex_exit(&_mutex);
434 return _error_remap(err);
435 }
436 file_descriptor[FILENO_INDEX(fd)].filesystem = fs;
437 strncpy(file_descriptor[FILENO_INDEX(fd)].path, path, PATH_MAX);
438
439 recursive_mutex_exit(&_mutex);
440
441 return _error_remap(fd);
442}

472// _write to STDOUT_FILENO and STDERR_FILENO falls back to `putchar` in the pico_stdio library
473static size_t pico_stdio_fallback_write(const void *buf, size_t nbyte) {
474 const char *out = buf;
475 for (size_t i = 0; i < nbyte; i++) {
476 putchar(out[i]);
477 }
478 stdio_flush();
479 return nbyte;
480}

669struct dirent *readdir(DIR *dir) {
670 auto_init_recursive_mutex(_mutex);
671 recursive_mutex_enter_blocking(&_mutex);
672
673 fs_dir_t *_dir = dir_descriptor[dir->fd].dir;
674 filesystem_t *fs = dir_descriptor[dir->fd].filesystem;
675 if (fs == NULL) {
676 _error_remap(-EBADF);
677 recursive_mutex_exit(&_mutex);
678 return NULL;
679 }
680 memset(&_dir->current, 0, sizeof(_dir->current));
681 int err = fs->dir_read(fs, _dir, &_dir->current);
682 if (err == 0) {
683 recursive_mutex_exit(&_mutex);
684 return &_dir->current;
685 } else if (err == -ENOENT) {
686 memset(&_dir->current, 0, sizeof(_dir->current));
687 _error_remap(0);
688 recursive_mutex_exit(&_mutex);
689 return NULL;
690 } else {
691 memset(&_dir->current, 0, sizeof(_dir->current));
692 _error_remap(err);
693 recursive_mutex_exit(&_mutex);
694 return NULL;
695 }

698char *fs_strerror(int errnum) {
699 if (errnum > 5000) {
700 // SD blockdevice error
701 const char *str = "";
