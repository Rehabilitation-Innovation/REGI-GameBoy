33static int fat_error_remap(FRESULT res) {
34 switch (res) {
35 case FR_OK: // (0) Succeeded
36 return 0;
37 case FR_DISK_ERR: // (1) A hard error occurred in the low level disk I/O layer
38 return -EIO;
39 case FR_INT_ERR: // (2) Assertion failed
40 return -1;
41 case FR_NOT_READY: // (3) The physical drive cannot work
42 return -EIO;
43 case FR_NO_FILE: // (4) Could not find the file
44 return -ENOENT;
45 case FR_NO_PATH: // (5) Could not find the path
46 return -ENOTDIR;
47 case FR_INVALID_NAME: // (6) The path name format is invalid
48 return -EINVAL;
49 case FR_DENIED: // (7) Access denied due to prohibited access or directory full
50 return -EACCES;
51 case FR_EXIST: // (8) Access denied due to prohibited access
52 return -EEXIST;
53 case FR_INVALID_OBJECT: // (9) The file/directory object is invalid
54 return -EBADF;
55 case FR_WRITE_PROTECTED: // (10) The physical drive is write protected
56 return -EACCES;
57 case FR_INVALID_DRIVE: // (11) The logical drive number is invalid
58 return -ENODEV;
59 case FR_NOT_ENABLED: // (12) The volume has no work area
60 return -ENODEV;
61 case FR_NO_FILESYSTEM: // (13) There is no valid FAT volume
62 return -EINVAL;
63 case FR_MKFS_ABORTED: // (14) The f_mkfs() aborted due to any problem
64 return -EIO;
65 case FR_TIMEOUT: // (15) Could not get a grant to access the volume within defined period
66 return -ETIMEDOUT;
67 case FR_LOCKED: // (16) The operation is rejected according to the file sharing policy
68 return -EBUSY;
69 case FR_NOT_ENOUGH_CORE: // (17) LFN working buffer could not be allocated
70 return -ENOMEM;
71 case FR_TOO_MANY_OPEN_FILES: // (18) Number of open files > FF_FS_LOCK
72 return -ENFILE;
73 case FR_INVALID_PARAMETER: // (19) Given parameter is invalid
74 return -EINVAL;
75 default:
76 return -res;
77 }
78}

114DRESULT disk_read(BYTE pdrv, BYTE *buff, LBA_t sector, UINT count) {
115 DWORD ssize = disk_get_sector_size(pdrv);
116 bd_size_t addr = (bd_size_t)sector * ssize;
117 bd_size_t size = count * ssize;
118 int err = _ffs[(int)pdrv]->read(_ffs[(int)pdrv], (const void *)buff, addr, size);
119 return err ? RES_PARERR : RES_OK;

122DRESULT disk_write(BYTE pdrv, const BYTE *buff, LBA_t sector, UINT count) {
123 debug_if(FFS_DBG, "disk_write [%d]\n", pdrv);
124 DWORD ssize = disk_get_sector_size(pdrv);
125 bd_size_t addr = (bd_size_t)sector * ssize;
126 bd_size_t size = count * ssize;
127
128 int err = _ffs[pdrv]->erase(_ffs[pdrv], addr, size);
129 if (err) {
130 return RES_PARERR;
131 }
132
133 err = _ffs[pdrv]->program(_ffs[pdrv], buff, addr, size);
134 if (err) {
135 return RES_PARERR;
136 }
137
138 return RES_OK;
139}

141DRESULT disk_ioctl(BYTE pdrv, BYTE cmd, void *buff) {
142 debug_if(FFS_DBG, "disk_ioctl(%d)\n", cmd);
143 switch (cmd) {
144 case CTRL_SYNC:
145 if (_ffs[pdrv] == NULL) {
146 return RES_NOTRDY;
147 } else {
148 return RES_OK;
149 }
150 case GET_SECTOR_COUNT:
151 if (_ffs[pdrv] == NULL) {
152 return RES_NOTRDY;
153 } else {
154 *((DWORD *)buff) = disk_get_sector_count(pdrv);
155 return RES_OK;
156 }
157 case GET_SECTOR_SIZE:
158 if (_ffs[pdrv] == NULL) {
159 return RES_NOTRDY;
160 } else {
161 *((WORD *)buff) = disk_get_sector_size(pdrv);
162 return RES_OK;
163 }
164 case GET_BLOCK_SIZE:
165 *((DWORD *)buff) = 1; // default when not known
166 return RES_OK;
167 case CTRL_TRIM:
168 if (_ffs[pdrv] == NULL) {
169 return RES_NOTRDY;
170 } else {
171 DWORD *sectors = (DWORD *)buff;
172 DWORD ssize = disk_get_sector_size(pdrv);
173 bd_size_t addr = (bd_size_t)sectors[0] * ssize;
174 bd_size_t size = (bd_size_t)(sectors[1] - sectors[0] + 1) * ssize;
175 int err = _ffs[pdrv]->trim(_ffs[pdrv], addr, size);
176 return err ? RES_PARERR : RES_OK;
177 }
178 }
179 return RES_PARERR;
180}

182DWORD get_fattime(void) {
183 time_t rawtime;
184 time(&rawtime);
185 struct tm *ptm = localtime(&rawtime);
186 return (DWORD)(ptm->tm_year - 80) << 25
187 | (DWORD)(ptm->tm_mon + 1) << 21
188 | (DWORD)(ptm->tm_mday) << 16
189 | (DWORD)(ptm->tm_hour) << 11
190 | (DWORD)(ptm->tm_min) << 5
191 | (DWORD)(ptm->tm_sec / 2);
192}

195static int mount(filesystem_t *fs, blockdevice_t *device, bool pending) {
196 filesystem_fat_context_t *context = fs->context;
197 mutex_enter_blocking(&context->_mutex);
198
199 char _fsid[3] = {0};
200 for (size_t i = 0; i < FF_VOLUMES; i++) {
201 if (_ffs[i] == NULL) {
202 context->id = i;
203 _ffs[i] = device;
204 _fsid[0] = '0' + i;
205 _fsid[1] = ':';
206 _fsid[2] = '\0';
207 FRESULT res = f_mount(&context->fatfs, _fsid, !pending);
208 mutex_exit(&context->_mutex);
209 return fat_error_remap(res);
210 }
211 }
212 mutex_exit(&context->_mutex);
213 return -ENOMEM;
214}

230static int format(filesystem_t *fs, blockdevice_t *device) {
231 filesystem_fat_context_t *context = fs->context;
232 mutex_enter_blocking(&context->_mutex_format);
233
234 if (!device->is_initialized) {
235 int err = device->init(device);
236 if (err) {
237 mutex_exit(&context->_mutex_format);
238 return err;
239 }
240 }
241
242 // erase first handful of blocks
243 bd_size_t header = 2 * device->erase_size;
244 int err = device->erase(device, 0, header);
245 if (err) {
246 mutex_exit(&context->_mutex_format);
247 return err;
248 }
249
250 size_t program_size = device->program_size;
251 void *buffer = malloc(program_size);
252 if (!buffer) {
253 mutex_exit(&context->_mutex_format);
254 return -ENOMEM;
255 }
256 memset(buffer, 0xFF, program_size);
257 for (size_t i = 0; i < header; i += program_size) {
258 err = device->program(device, buffer, i, program_size);
259 if (err) {
260 free(buffer);
261 mutex_exit(&context->_mutex_format);
262 return err;
263 }
264 }
265 free(buffer);
266
267 // trim entire device to indicate it is unneeded
268 err = device->trim(device, 0, device->size(device));
269 if (err) {
270 mutex_exit(&context->_mutex_format);
271 return err;
272 }
273
274 err = fs->mount(fs, device, true);
275 if (err) {
276 mutex_exit(&context->_mutex_format);
277 return err;
278 }
279
280 MKFS_PARM opt;
281 opt.fmt = (FM_ANY | FM_SFD);
282 opt.n_fat = 0;
283 opt.align = 0U;
284 opt.n_root = 0U;
285 opt.au_size = 0; // (DWORD)cluster_size;
286
287 char id[3] = "0:";
288 id[0] = '0' + context->id;
289
290 FRESULT res = f_mkfs((const TCHAR *)id, &opt, NULL, FF_MAX_SS);
291
292 if (res != FR_OK) {
293 fs->unmount(fs);
294 mutex_exit(&context->_mutex_format);
295 return fat_error_remap(res);
296 }
297
298 err = fs->unmount(fs);
299 if (err) {
300 mutex_exit(&context->_mutex_format);
301 return res;
302 }
303
304 mutex_exit(&context->_mutex_format);
305 return 0;
306}

308static const char *fat_path_prefix(char *dist, int id, const char *path) {
309 if (id == 0) {
310 strcpy(dist, path);
311 return (const char *)dist;
312 }
313
314 dist[0] = '0' + id;
315 dist[1] = ':';
316 strcpy(dist + strlen("0:"), path);
317 return (const char *)dist;

386static int file_stat(filesystem_t *fs, const char *path, struct stat *st) {
387 filesystem_fat_context_t *context = fs->context;
388 char fpath[PATH_MAX];
389 fat_path_prefix(fpath, context->id, path);
390 FILINFO f = {0};
391
392 mutex_enter_blocking(&context->_mutex);
393 FRESULT res = f_stat(fpath, &f);
394 mutex_exit(&context->_mutex);
395
396 if (res != FR_OK) {
397 return fat_error_remap(res);
398 }
399 st->st_size = f.fsize;
400 st->st_mode = 0;
401 st->st_mode |= (f.fattrib & AM_DIR) ? S_IFDIR : S_IFREG;
402 st->st_mode |= (f.fattrib & AM_RDO) ?
403 (S_IRUSR | S_IXUSR | S_IRGRP | S_IXGRP | S_IROTH | S_IXOTH) :
404 (S_IRWXU | S_IRWXG | S_IRWXO);
405 return 0;
406}

408static int file_open(filesystem_t *fs, fs_file_t *file, const char *path, int flags) {
409 BYTE open_mode;
410 if (flags & O_RDWR)
411 open_mode = FA_READ | FA_WRITE;
412 else if (flags & O_WRONLY)
413 open_mode = FA_WRITE;
414 else
415 open_mode = FA_READ;
416 if (flags & O_CREAT) {
417 if (flags & O_TRUNC)
418 open_mode |= FA_CREATE_ALWAYS;
419 else
420 open_mode |= FA_OPEN_ALWAYS;
421 }
422 if (flags & O_APPEND)
423 open_mode |= FA_OPEN_APPEND;
424
425 char fpath[PATH_MAX];
426 filesystem_fat_context_t *context = fs->context;
427 fat_path_prefix(fpath, context->id, path);
428 fat_file_t *fat_file = file->context = calloc(1, sizeof(fat_file_t));
429 if (fat_file == NULL) {
430 fprintf(stderr, "file_open: Out of memory\n");
431 return -ENOMEM;
432 }
433
434 mutex_enter_blocking(&context->_mutex);
435 FRESULT res = f_open(&fat_file->file, fpath, open_mode);
436 mutex_exit(&context->_mutex);
437
438 if (res != FR_OK) {
439 debug_if(FFS_DBG, "f_open('w') failed: %d\n", res);
440 return fat_error_remap(res);
441 }
442 return 0;
443}

517 (void)fs;

519 fat_file_t *fat_file = file->context;
520
521 mutex_enter_blocking(&context->_mutex);

532 return fat_error_remap(res);
