34static void print_progress(const char *label, size_t current, size_t total) {
35 int num_dots = (int)((double)current / total * (50 - strlen(label)));
36 int num_spaces = (50 - strlen(label)) - num_dots;
37
38 printf("\r%s ", label);
39 for (int i = 0; i < num_dots; i++) {
40 printf(".");
41 }
42 for (int i = 0; i < num_spaces; i++) {
43 printf(" ");
44 }
45 printf(" %zu/%zu bytes", current, total);
46 fflush(stdout);
47}

66static uint32_t xor_rand(uint32_t *seed) {
67 *seed ^= *seed << 13;
68 *seed ^= *seed >> 17;
69 *seed ^= *seed << 5;

77static void benchmark_write(void) {
78 const char *label = "Write";
79 absolute_time_t start_at = get_absolute_time();
80
81 int fd = open("/benchmark", O_WRONLY|O_CREAT);
82 if (fd == -1) {
83 printf("open error: %s\n", strerror(errno));
84 return;
85 }
86
87 uint32_t counter = 0;
88 xor_rand(&counter);
89 uint8_t buffer[BUFFER_SIZE] = {0};
90 size_t remind = BENCHMARK_SIZE;
91 while (remind > 0) {
92 size_t chunk = remind % sizeof(buffer) ? remind % sizeof(buffer) : sizeof(buffer);
93 uint32_t *b = (uint32_t *)buffer;
94 for (size_t j = 0; j < (chunk / sizeof(uint32_t)); j++) {
95 b[j] = xor_rand_32bit(&counter);
96 }
97
98 ssize_t write_size = write(fd, buffer, chunk);
99 if (write_size == -1) {
100 printf("write: error: %s\n", strerror(errno));
101 return;
102 }
103 remind = remind - write_size;
104 print_progress(label, BENCHMARK_SIZE - remind, BENCHMARK_SIZE);
105 }
106
107 int err = close(fd);
108 if (err == -1) {
109 printf("close error: %s\n", strerror(errno));
110 return;
111 }
112
113 double duration = (double)absolute_time_diff_us(start_at, get_absolute_time()) / 1000 / 1000;
114 printf(" %.1f KB/s\n", (double)(BENCHMARK_SIZE) / duration / 1024);
115}

161int main(void) {
162 stdio_init_all();
163 init_filesystem_combination();
164
165 for (size_t i = 0; i < NUM_COMBINATION; i++) {
166 struct combination_map setting = combination[i];
167 printf("Test of %s on %s:\n", setting.filesystem->name, setting.device->name);
168
169 int err = fs_format(setting.filesystem, setting.device);
170 if (err == -1 && errno == 5005) {
171 printf("skip, device not connected\n");
172 continue;
173 }
174 if (err == -1) {
175 printf("fs_format error: %s\n", fs_strerror(errno));
176 return -1;
177 }
178 err = fs_mount("/", setting.filesystem, setting.device);
179 if (err == -1) {
180 printf("fs_mount / error: %s\n", fs_strerror(errno));
181 return -1;
182 }
183
184 benchmark_write();
185 benchmark_read();
186
187 err = fs_unmount("/");
188 if (err == 01) {
189 printf("fs_unmount / error: %s\n", fs_strerror(errno));
190 return -1;
191 }
192 }
193 printf(COLOR_GREEN("ok\n"));
194}
