26static int init(blockdevice_t *device) {
27 (void)device;
28 blockdevice_loopback_config_t *config = device->config;
29 mutex_enter_blocking(&config->_mutex);
30 if (device->is_initialized) {
31 mutex_exit(&config->_mutex);
32 return BD_ERROR_OK;
33 }
34
35 config->fildes = open(config->path, O_RDWR|O_CREAT, 0644);
36 if (config->fildes == -1 && errno == EEXIST) {
37 config->fildes = open(config->path, O_RDWR);
38 }
39 if (config->fildes == -1) {
40 mutex_exit(&config->_mutex);
41 return -errno;
42 }
43
44 device->is_initialized = true;
45 mutex_exit(&config->_mutex);
46 return BD_ERROR_OK;
47}

49static int deinit(blockdevice_t *device) {
50 blockdevice_loopback_config_t *config = device->config;
51 mutex_enter_blocking(&config->_mutex);
52
53 if (!device->is_initialized) {

72static int __read(blockdevice_t *device, const void *buffer, bd_size_t addr, bd_size_t length) {
73 blockdevice_loopback_config_t *config = device->config;
74 mutex_enter_blocking(&config->_mutex);
75
76 off_t off = lseek(config->fildes, addr, SEEK_SET);
77 if (off == -1) {
78 mutex_exit(&config->_mutex);
79 return -errno;
80 }
81
82 ssize_t read_size = read(config->fildes, (void *)buffer, (size_t)length);
83 if (read_size == -1) {
84 mutex_exit(&config->_mutex);
85 return -errno;
86 }
87 if ((size_t)read_size < (size_t)length) {
88 size_t remind = length - read_size;
89 memset((void *)buffer + read_size, 0, remind);
90 }
91
92 mutex_exit(&config->_mutex);
93 return BD_ERROR_OK;
94}

134blockdevice_t *blockdevice_loopback_create(const char *path, size_t capacity, size_t block_size) {
135 blockdevice_t *device = calloc(1, sizeof(blockdevice_t));
136 if (device == NULL) {
137 return NULL;
138 }
139 blockdevice_loopback_config_t *config = calloc(1, sizeof(blockdevice_loopback_config_t));
140 if (config == NULL) {
141 free(device);
142 return NULL;
143 }
144
145 device->init = init;
146 device->deinit = deinit;
147 device->read = __read;
148 device->erase = erase;
149 device->program = program;
150 device->trim = trim;
151 device->sync = __sync;
152 device->size = size;
153 device->read_size = block_size;
154 device->erase_size = block_size;
155 device->program_size = block_size;
156 device->name = DEVICE_NAME;
157 device->is_initialized = false;
158
159 config->path = path;
160 config->capacity = capacity;
161 config->block_size = block_size;
162 config->fildes = -1;
163
164 mutex_init(&config->_mutex);
165 device->config = config;
166 if (device->init(device) != BD_ERROR_OK) {
167 free(config);
168 free(device);
169 return NULL;
170 }
171 return device;
172}
